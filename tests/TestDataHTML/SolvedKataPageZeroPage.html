    Human Readable Time
    TypeScript:
    export function humanReadable(sourceSeconds:number):string {
  let remainder = sourceSeconds

  const hours = Math.floor(remainder / 3600)
  remainder = remainder % 3600

  const minutes = Math.floor(remainder / 60)
  const seconds = remainder % 60

  return [hours, minutes, seconds]
    .map((part) => part.toString().padStart(2, '0'))
    .join(':')
}

    function sumStrings(a,b) {
  a = [...a.replace(/^0+/, '')].map(Number).reverse();
  b = [...b.replace(/^0+/, '')].map(Number).reverse();

  if(a.length < b.length) {
    [a, b] = [b, a];
  }

  let l = a.length;
  let result = new Array(l + 1);
  let over = 0;

  for(let i = 0; i < l; i++) {
    let s = a[i] + over;
    if(i < b.length) s+= b[i];
    result[i] = s % 10;
    over = (s >= 10) ? 1 : 0;
  }

  if(over) {
    result[result.length - 1] = over;
  }

  return result.reverse().join('');
}

    Find the unique string
    JavaScript:
    function findUniq(arr) {
  const getChars = (word) => Array.from(new Set(word.toLowerCase())).sort().join('').trim()
  let sample = arr.slice(0, 3).map(getChars).sort();
  let same = (sample[0] == sample[1]) ? sample[0] : sample[2];

  for(let word of arr) {
    if(getChars(word) !== same)
      return word;
  }
}
