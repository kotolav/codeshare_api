function upArray(arr){
  if(arr.length === 0) return null
  let add = 1, res = []
  for(let i = arr.length - 1; i >= 0; i--) {
    let e = arr[i]
    if(e < 0 || e > 10) return null
    let mod = (e + add) % 10
    add = Math.floor((e + add) / 10)
    res.unshift(mod)
  }
  if(add > 0)
    res.unshift(add)
  return res
}

function formatWords(words){
  if(!words) return '';
  words = words.filter(w => w.length !== 0);
  if(words.length === 0) return '';
  return (words.length > 1) ? [[...words.slice(0, -1)].join(', '), ...words.slice(-1)].join(' and ') : words[0];

function scramble(str1, str2) {
  let dict = Array.from(str1).reduce((acc, el) => {
    acc[el] ? acc[el]++ : acc[el] = 1
    return acc
  }, {})

  for(let c of str2) {
    if(dict[c] && dict[c] > 0) {
      dict[c]--
    } else {
      return false
    }
  }

  return true
}

function scramble(str1, str2) {
  let counter = {}
  for(let c of str2) {
    counter[c] = counter[c] + 1 || 1;
  }

  for(let c of str1) {
    if(counter[c]) {
      counter[c]--;
      if(!counter[c]) {
        delete counter[c];
        if(Object.keys(counter).length == 0)
          return true;
      }
    }
  }

  return false;
}

function matrixAddition(a, b){
  let sum = Array.from({length: a.length}, () => new Array(a.length).fill(0));
  for(let x = 0, l = a.length; x < l; x++) {
    for (let y = 0, ll = a[x].length; y < ll; y++) {
      sum[x][y] = a[x][y] + b[x][y]
    }
  }

  return sum
}

class Inspector {
  constructor() {
    this.rules = {'documents' : {}, 'access': new Set(), 'vaccination': {}}
    this.currentDate = new Date('1982.11.22')
  }
  receiveBulletin(string) {
    let data = string.split('\n')
    this.rules['documents'] = this.parseDocuments(data, this.rules['documents'])
    this.rules['vaccination'] = this.parseVaccination(data, this.rules['vaccination'])
    this.rules['access'] = this.parseAccess(data, this.rules['access'])
    this.rules['criminal'] = [this.parseCriminal(string)]
    this.currentDate.setDate(this.currentDate.getDate() + 1)
  }

  parseCriminal(strings) {
    let parsed = strings.match(/Wanted by the State: (.*?) (.*?)$/i)
    if(parsed) {
      return parsed[2] + ', ' + parsed[1]
    }
    return ''
  }

  parseDocuments(strings, obj) {
    let rules = obj
    let data = strings.filter(x => x.includes('require') && !x.includes('vaccination'))
    for(let s of data) {
      let parsed = s.match(/^(?:Citizens of )?(.*?) (no longer )?require (.*?)$/i)
      let kind = parsed[1].replace(/_/g, ' ')
      for(let k of kind.split(', ')) {
        if(parsed[2]) {
          delete rules[k]
        } else {
          rules[k] = parsed[3]
        }
      }
    }
    return rules
  }

  parseVaccination(strings, obj) {
    let rules = obj
    let data = strings.filter(x => x.includes('vaccination'))
    for(let s of data) {
      let parsed = s.match(/^(?:Citizens of )?(.*?) (no longer )?require (.*?)$/i)
      let kind = parsed[1].replace(/_/g, ' ')
      let value = parsed[3].replace(' vaccination', '')
      for(let country of kind.split(', ')) {
        if(parsed[2]) {
          delete rules[country]
        } else {
          rules[country] = value
        }
      }
    }
    return rules
  }

  parseAccess(strings, obj) {
    let rules = obj
    let data = strings.filter(x => /\b(allow|deny)\b/i.test(x))
    for(let s of data) {
      let parsed = s.match(/\b(allow|deny) citizens of (.*?)$/i)
      let kind = parsed[1].toLowerCase()
      let newCountries = parsed[2].split(', ')
      switch(kind) {
        case 'allow': rules = new Set([...rules, ...newCountries]); break
        case 'deny': rules = new Set([...rules].filter(x => !newCountries.includes(x))); break
      }
    }
    return rules
  }

  parseValue(str, parseKey, obj, objKey) {
    let result = str.match(new RegExp(parseKey + ': (.*)'))
    if(result)
      obj[objKey] = result[1]
  }

  parsePapers(data) {
    let papers = {}
    for(let [document, str] of Object.entries(data))
    {
      let info = {}
      let reg = new RegExp('(.*): (.*)', 'g')
      let result
      while((result = reg.exec(str)) !== null) {
        let key = result[1].replace(/_/g, ' ').toLowerCase()
        if(key == 'id#') key = 'id'
        info[key] = result[2]
      }
      papers[document.replace(/_/g, ' ')] = info
    }
    return papers
  }

  checkNation(nation) {
    if(this.rules.access.has(nation))
      return true
    throw('Entry denied: citizen of banned nation.')
  }

  checkCriminal(name) {
    if(this.rules.criminal.includes(name))
      throw('Detainment: Entrant is a wanted criminal.')
  }

  checkExpiration(person) {
    for(let [paper, data] of Object.entries(person)) {
      if(data.exp && ((new Date(data.exp)) < (this.currentDate))) {
        throw ('Entry denied: ' + paper.replace(/_/g, ' ') + ' expired.')
      }
    }
  }

  checkVaccination(person) {
    if(this.rules.vaccination) {
      for(let personType of person.info['personType']) {
        if(personType in this.rules.vaccination) {
          if(!(person['papers']['certificate of vaccination'] && person['papers']['certificate of vaccination']['vaccines'].includes(this.rules.vaccination[personType]))) {
            throw('Entry denied: missing required certificate of vaccination.')
          }
        }
      }
    }
  }

  makeInfo(person) {
    let info = {}
    for(let [paper, paperData] of Object.entries(person)) {
      for(let [key, data] of Object.entries(paperData)) {
        if(['exp'].includes(key)) continue
        if(key == 'name' && this.rules.criminal.includes(data))
          throw('Detainment: Entrant is a wanted criminal.')
        if((key in info) && (info[key] !== data)) {
          let errKey = key

          switch(key) {
            case 'id': errKey = 'ID number'; break
            case 'nation': errKey = 'nationality'; break
          }

          throw('Detainment: ' + errKey + ' mismatch.')
        } else {
          info[key.replace(/_/g, ' ')] = data
        }
      }
    }
    info['personType'] = new Set(['Entrants'])
    if(info['nation'] && info['nation'].toLowerCase() == 'arstotzka')
      info['personType'].add('Arstotzka')
    else
      info['personType'].add('Foreigners').add(info['nation'])
    if(info['purpose'] && (info['purpose'].toLowerCase() == 'work'))
      info['personType'].add('Workers')
    return info
  }

  checkRequirements(person) {
    for(let [personType, paper] of Object.entries(this.rules.documents)) {
      if(person.info.personType.has(personType)) {
        if(paper == 'certificate of vaccination') continue
        if(!((paper in person.papers)
            || ((paper == 'access permit') && ('grant of asylum' in person.papers))
            )) {
          if((paper == 'access permit') && ('diplomatic authorization' in person.papers))
          {
            if(!person.papers['diplomatic authorization']['access'].includes('Arstotzka'))
            {
              throw('Entry denied: invalid diplomatic authorization.')
            } else {
              continue
            }
          }
          throw('Entry denied: missing required ' + paper + '.')
        }
      }
    }
  }

  inspect(strings) {
    let person = {'papers': this.parsePapers(strings)}
    try {
      person = Object.assign(person, {'info': this.makeInfo(person.papers)})
      this.checkExpiration(person.papers)
      this.checkRequirements(person)
      this.checkVaccination(person)
      this.checkNation(person.info.nation)
    } catch(err) {
      return err
    }

    return person.info.nation == 'Arstotzka' ? 'Glory to Arstotzka.' : 'Cause no trouble.'
  }
}

  let s = str.split(' ').reduce((s, w) => s + w.charAt(0).toUpperCase() + w.slice(1).toLowerCase(), '')

function phone(strng, num) {
    let strings = strng.split('\n')
    let db = {}
    for (let person of strings) {
      let phone, address, name
      person = person.replace(/\+?([\-\d]{11,})/, function(s, p1) {
        phone = p1
        return ''
      })
      person = person.replace(/<(.*?)>/, function(s, p1) {
        name = p1
        return ''
      })
      address = (person.replace(/[^\w\s\-\.]+/g, '').replace(/(\s{2,}|_)/g, ' ')).trim()
      db[phone] = phone in db ? {'error': 'Too many people: ' + phone} : {'phone': phone, 'name': name, 'address': address}
    }

    let found = db[num]
    if(!found)
      return 'Error => Not found: ' + num
    if('error' in found)
      return 'Error => ' + found['error']
    else
      return `Phone => ${found['phone']}, Name => ${found['name']}, Address => ${found['address']}`
}
